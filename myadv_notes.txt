table:  see ct_gprmc.sql

how to get lat + long: 
select
  round (   trunc ( g.lat / 100 ) 
         +  mod ( g.lat / 100, 1 ) * 100/60, 6  ) as lat_dec
, ' ,  ' as komma 
, round (   trunc ( g.lon / 100 ) 
         +  mod ( g.lon / 100, 1 ) * 100/60, 6 ) as lon_dec
-- , g.* 
from gprmc g 
order by g.utc_hhmmss ;
